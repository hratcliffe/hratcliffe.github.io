231 void controller::bounce_average(bounce_av_data bounce_dat){
238  if(spect_D_list.size() == 0) return;
239  //Empty list, nothing to do.
240 
241  //Add new averaged D
242  size_t dims[2];
243  get_D_size(dims);
244  this->add_d_special(dims[0], dims[1]);
245  //Tag as bounce av'd and copy block id from block used
246  d_specials_list[current_special_d]->tag = BOUNCE_AV;
247  strcpy(d_specials_list[current_special_d]->block_id, spect_D_list[current_pair].second->block_id);
248 
249  //Flatten down each d onto this one with all integrand included
250  bool mirror_block = false;//Flag for the space block which contains mirror point
251  my_type val, current_lat = 0.0, d_lat = 0.0, D_val, lat_factor, mirror_lat = pi/2.0, include_fraction, alpha_current, used_lat;
252  size_t n_blocks = spect_D_list.size();
253  my_type sin_theta, cos_theta_sq, alpha_eq;
254  //Latitude increment block to block in RADIANS
255  d_lat = bounce_dat.max_latitude*pi/180.0/(my_type) n_blocks;
256  //Loop over every p and alpha_eq in result D array
257  for(size_t p_i = 0; p_i < dims[0]; p_i++){
258  for(size_t ang_i = 0; ang_i < dims[1]; ang_i++){
259  alpha_eq = spect_D_list[0].second->get_axis_element_ang(ang_i);//in RADIANS
260  val = 0.0;
261  //Initial block-centred latitude
262  current_lat = d_lat/2.0;//In RADIANS
263  if(bounce_dat.type != plain){
264  //For plain we ignore mirror lat and use 90=pi/2
265  //For others we integrate up to mirror_lat
266  mirror_lat = solve_mirror_latitude(alpha_eq);
267  }
268  for(size_t block_i = 0; block_i < n_blocks; block_i++){
269  //Don't include cells above mirror lat at all
270  mirror_block = (current_lat + d_lat/2.0 >= mirror_lat);
271  //For block containing mirror lat we include only the part up to the actual inclusion
272  if(current_lat - d_lat/2.0 >= mirror_lat) continue;
273  sin_theta = std::sin(pi/2.0 - current_lat);
274  cos_theta_sq = 1.0 - sin_theta*sin_theta;
275 
276  //This one is a simple case
277  if(bounce_dat.type == plain){
278  lat_factor = bounce_dat.L_shell * R_E * std::sqrt(1.0+3.0*cos_theta_sq) * sin_theta * d_lat;//This is ds as in Schulz/Lanzerotti Eq 1.18.
279  if(mirror_block){
280  //We now know mirror is between current_lat Â± d_lat/2.0
281  //Calculate how much of cell to include
282  include_fraction = -(current_lat - d_lat/2.0 - mirror_lat)/d_lat;
283  }else{
284  include_fraction = 1.0;
285  }
286  D_val = spect_D_list[block_i].second->get_element(p_i, ang_i);//Value of D
287  val += D_val * lat_factor*include_fraction;
288  //Sum up the blocks done on this processor
289  current_lat += d_lat;
290 
291  }else{
292  //The other cases we need the actual alpha and the rest of the integrand
293  //Get the correct alpha to read D etc at
294  if(mirror_block){
295  used_lat = (current_lat-d_lat/2.0 +mirror_lat)/2.0;
296  include_fraction = (mirror_lat - (current_lat-d_lat/2.0))/d_lat;
297  //Because we can't simply solve alpha at middle lat as this might already be out of range.
298  }else{
299  used_lat = current_lat;
300  include_fraction = 1.0;
301  }
302  alpha_current = alpha_from_alpha_eq(alpha_eq, used_lat);
303 
304  //Calculate the latitude part of the integrand
305  my_type c7_lat = std::pow(cos(current_lat), 7);
306  switch(bounce_dat.type){
307  case plain:
308  //Already handled above
309  break;
310  case alpha_alpha:
311  lat_factor = cos(alpha_current)*c7_lat/std::pow(cos(alpha_eq), 2);
312  break;
313  case alpha_p:
314  case p_alpha:
315  lat_factor = sin(alpha_current)*c7_lat/(cos(alpha_eq)*sin(alpha_eq));
316  break;
317  case p_p:
318  lat_factor = std::pow(sin(alpha_current)/sin(alpha_eq), 2)*c7_lat/cos(alpha_current);
319  break;
320  }
321  //Get D and add contribution on this block
322  D_val = spect_D_list[block_i].second->get_element_by_values(spect_D_list[block_i].second->get_axis_element(0, p_i), alpha_current);
323 
324  val += D_val * lat_factor*include_fraction*d_lat;
325  current_lat += d_lat;
326  }
327  }
328  //Now set the correct element
329  if(bounce_dat.type != plain){
330  d_specials_list[current_special_d]->set_element(p_i, ang_i, val/bounce_period_approx(alpha_eq));
331  }else{
332  d_specials_list[current_special_d]->set_element(p_i, ang_i, val);
333  }
334  }
335  }
336  handle_d_mpi();
337  //Now mpi reduce the results to get a global average
338 
339 }

341 void controller::handle_d_mpi(){
347 //Make a reduced d on root, but leaving the current_special_d counter unchanged.
348  size_t dims[2];
349  get_D_size(dims);
350  int total = dims[0]*dims[1];
351 
352  int prev_current_d = current_special_d;
353  if(mpi_info.rank ==0){
354  //Add a new global value on root only
355  add_d_special(dims[0], dims[1]);
356  //Copy roots bounce average into this
357  *(d_specials_list[current_special_d]) = *(d_specials_list[prev_current_d]);
358  d_specials_list[current_special_d]->tag = GLOBAL;
359  }
360  //reset current_special_d (affects root only)
361  int root_global_d = current_special_d;
362  current_special_d = prev_current_d;
363 
364  //Reduce onto this new root d so we preserve all the local versions too
365  MPI_Reduce(get_current_d()->data, d_specials_list[root_global_d]->data , total, MPI_MYTYPE, MPI_SUM, 0, MPI_COMM_WORLD);
366 
367 }

428 my_type solve_mirror_latitude(my_type alpha_eq, bool print_iters){
436  my_type min_alpha = 0.00001;//Minimum pitch angle to attempt solution. Below this we default to pi. Note this is a mirror latitude of over 89.2 degrees
437  if(std::abs(alpha_eq) < min_alpha) return pi/2;
438 
439  my_type last_solution, next_solution;
440  //Initial guess is taken using approximation of L^6 = 4 s4alpha
441  last_solution = 1.26 *std::pow(sin(alpha_eq), 2/3.0);
442 
443  size_t max_iter = 20;//Maximum iterations to try
444  my_type precision = 1e-3*pi/100.0;//precision in solution to stop at. this is 0.1% of pi i.e. 0.2 deg
445 
446  my_type s4alpha = std::pow(sin(alpha_eq), 4);
447  for(size_t iter = 0; iter < max_iter; iter++){
448  if(print_iters) my_print(mk_str(iter)+' '+mk_str(last_solution)+' '+mk_str(acos(std::sqrt(last_solution))*180.0/pi), mpi_info.rank);
449  next_solution = Newton_Raphson_iteration(last_solution, s4alpha);
450  if(std::abs(acos(std::sqrt(last_solution)) - acos(std::sqrt(next_solution))) < precision) break;
451  last_solution = next_solution;
452  }
453 
454  return acos(std::sqrt(next_solution));
455 }
