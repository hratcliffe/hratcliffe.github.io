292 calc_type lookup(calc_type p_par, calc_type p_perp, my_type * data, size_t par_sz, size_t perp_sz,calc_type dp_par_ax, calc_type p_par_ax_min, calc_type dp_perp_ax, calc_type p_perp_ax_min){
312  calc_type p_par_ind_decimal, p_perp_ind_decimal;
313  long long p_par_ind, p_perp_ind;
314  calc_type f_tmp_plus, f_tmp_minus, par_diff, perp_diff, interp_val;
315  //Calculate axis index for par and perp
316  if(p_par > p_par_ax_min){
317  p_par_ind_decimal = (p_par - p_par_ax_min)/dp_par_ax;
318  p_par_ind = std::floor(p_par_ind_decimal);
319  }else{
320  p_par_ind = -1;
321  }
322  if(p_perp > p_perp_ax_min){
323  p_perp_ind_decimal = (p_perp - p_perp_ax_min)/dp_perp_ax;
324  p_perp_ind = std::floor( p_perp_ind_decimal);
325  }else{
326  p_perp_ind = -1;
327  }
328  //Check we're in range of the table
329  //Casts here should only ever widen the type
330  if(p_par_ind >= (long long) par_sz || p_perp_ind >= (long long)perp_sz || p_par_ind <0 || p_perp_ind < 0){
331  //If out of range, assume 0. Would also be reasonable to assume last in-range value.
332  return 0.0;
333  }
334  //Calculate the fraction of cell for interpolation
335  par_diff = p_par_ind_decimal - p_par_ind;
336  perp_diff = p_perp_ind_decimal - p_perp_ind;
337  //4 point linear interpolation to "exact" location
338  //Calculate f at lower and upper perp values interpolated on par lines
339  f_tmp_minus = par_diff *data[par_sz*p_perp_ind + p_par_ind + 1] + (1.0 - par_diff)* data[par_sz*p_perp_ind + p_par_ind];
340  f_tmp_plus = par_diff *data[par_sz*(p_perp_ind + 1) + p_par_ind + 1] + (1.0 - par_diff)* data[par_sz*(p_perp_ind + 1) + p_par_ind];
341 
342  interp_val = (perp_diff * f_tmp_plus + (1.0 - perp_diff)*f_tmp_minus);
343 
344  return interp_val;
