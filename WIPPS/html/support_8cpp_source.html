546 bool flatten_fortran_slice(my_type * src_ptr, my_type* dest_ptr, size_t n_dims_in, size_t * dims_in, size_t flatten_on_dim, size_t flat_start, size_t flat_stop){
559 /*
560 * A 2-d array 5x3 is
561 |ooooo||ooooo||ooooo|
562 <-row->
563 
564 A 3-d 5x3x2 is
565 [|ooooo||ooooo||ooooo|][|ooooo||ooooo||ooooo|]
566 <--------'slice'------>
567 
568 We'll borrow the resizer code to get the new array by resizing the required dim to size 1 in a copy, and then we'll add the rest on
569 Think of the array as being 3-D. The dim we;re flattening is dim-1. All the less-significant dims are smushed together into dim-0, and all the higher into dim-2
570 */
571 
572  if(flatten_on_dim > n_dims_in) return 1;
573  if(flat_start >= dims_in[flatten_on_dim]) return 1;
574  if(flat_stop >= dims_in[flatten_on_dim]) flat_stop = dims_in[flatten_on_dim];
575 
576  size_t els_to_copy = 1, n_segments = 1, sz = 1;
577 
578  for(size_t i=0; i<flatten_on_dim; ++i) els_to_copy*= dims_in[i];
579  for(size_t i=flatten_on_dim+1; i< n_dims_in; ++i) n_segments *= dims_in[i];
580 
581  size_t chunk_sz = els_to_copy;
582  for(size_t i=0; i< n_segments; ++i) std::copy(src_ptr + i*chunk_sz*dims_in[flatten_on_dim] +els_to_copy*flat_start, src_ptr + i*chunk_sz*dims_in[flatten_on_dim]+ els_to_copy+ els_to_copy*flat_start, dest_ptr + i*chunk_sz*sz);
583 
584  //Now we should have the 0th row in place
585  //Add each successive row onto it
586 
587  for(size_t j = flat_start+1; j<flat_stop; j++){
588 
589  for(size_t i=0; i< n_segments; ++i) std::transform(src_ptr + i*chunk_sz*dims_in[flatten_on_dim] + els_to_copy*j, src_ptr + i*chunk_sz*dims_in[flatten_on_dim]+ els_to_copy + els_to_copy*j, dest_ptr + i*chunk_sz*sz,dest_ptr + i*chunk_sz*sz, std::plus<my_type>());
590  }
591 
592  return 0;
593 
594 }
