288 mu_dmudom plasma::get_phi_mu_from_stix(calc_type w, calc_type psi, calc_type alpha, int n, calc_type gamma_particle, calc_type R, calc_type L, calc_type P, bool skip_phi, bool Righthand)const{
303  calc_type w2, w3;
304  w2 = w*w;
305  w3 = w2*w;
306 
307  calc_type J, S, D, A, B, C, s2psi, c2psi, mua2, mub2, mu2, scpsi;
308  calc_type F, G, smu;
309  calc_type dHdF, dHdG, dmudw, dFdpsi,dGdpsi, dAdpsi,dBdpsi;
310  calc_type wp[ncomps], wp2[ncomps], wc[ncomps], X[ncomps], Y[ncomps];
311 
312  calc_type dmudX[ncomps], dmudY[ncomps];
313  calc_type dPdX[ncomps], dLdX[ncomps], dRdX[ncomps], dSdX[ncomps];
314  calc_type dAdX[ncomps], dBdX[ncomps], dCdX[ncomps], dFdX[ncomps], dGdX[ncomps];
315  calc_type dLdY[ncomps], dRdY[ncomps], dSdY[ncomps];
316  calc_type dAdY[ncomps], dBdY[ncomps], dCdY[ncomps], dFdY[ncomps], dGdY[ncomps], dXdw[ncomps], dYdw[ncomps];
317 
318  //These will hold suitably calc'd plasma frequency, square and cyclotron freq. If we have to derive from position, we do...
319  for(int i=0; i<ncomps; ++i){
320  wp[i] = sqrt(pdens[i] * pcharge[i]*pcharge[i]/(eps0 * pmass[i]));
321  wp2[i] = wp[i]*wp[i];
322  wc[i] = (pcharge[i]) * this->B0 / pmass[i];
323 
324  X[i] = wp2[i]/w2;
325  Y[i] = wc[i]/w;
326  }
327 
328 //We loop over components and trust compiler to unroll for us :) most of these will trivially vectorise anyway.
329 
330  scpsi = std::sin(psi);
331  s2psi = std::pow(scpsi, 2);
332  c2psi = 1.0 - s2psi;
333  scpsi *= std::sqrt(c2psi);
334  //To make it smokin'
335 
336  S = 0.5*(R + L);
337  D = 0.5*(R - L);
338  A = S*s2psi + P*c2psi;
339  B = R*L*s2psi + P*S*(1.0+c2psi);
340  C = P*R*L;
341  J = std::sqrt(B*B - 4.0*A*C);
342 
343  mua2 = 1.0 - 2.0*(A - B + C)/(2.0*A - B + J);
344  mub2 = 1.0 - 2.0*(A - B + C)/(2.0*A - B - J);
345 
346  mu_dmudom my_mu;
347  //placeholder values if we can't fill...
348  my_mu.mu = 1.0;
349  my_mu.dmudom = 0.0;
350  my_mu.dmudtheta = 0.0;
351  my_mu.err = 1;
352  my_mu.phi = 0;
353  my_mu.cone_ang = pi;
354  smu = 0;
355  mu2 = 0;
356 
357  if( (mua2 > 0.0) || (mub2 > 0.0) ){
358  //see Albert [2005] or Stix [1972] for mode selection
359  if(Righthand){//Select Mode
360  if(w > 0){
361  if(D < 0.0){ smu = 1.0; mu2 = mua2;}
362  else{smu = -1.0; mu2 = mub2;}
363  }else{
364  if(D < 0.0){smu = -1.0; mu2 = mub2;}
365  else{smu = 1.0; mu2 = mua2;}
366  }
367  }else{
368  if(w > 0){
369  if(D > 0.0){ smu = 1.0; mu2 = mua2;}
370  else{smu = -1.0; mu2 = mub2;}
371  }else{
372  if(D > 0.0){smu = -1.0; mu2 = mub2;}
373  else{smu = 1.0; mu2 = mua2;}
374  }
375  }
376 
377  my_mu.mu = std::sqrt(mu2);
378  my_mu.err = 0;
379 
380  F = 2.0*(A - B + C);
381  G = 2.0*A - B + smu*J;
382  dHdF = -1.0/G;
383  dHdG = F/(G*G);
384 
385  for(int i=0; i<ncomps; i++){
386  dPdX[i] = -1.0;
387  dLdX[i] = -1.0/(1.0 - Y[i]);
388 
389  dRdX[i] = -1.0/(1.0 + Y[i]);
390  dSdX[i] = 0.5*(dRdX[i] + dLdX[i]);
391  dAdX[i] = 0.5*(dRdX[i] + dLdX[i])*s2psi + dPdX[i]*c2psi;
392  dBdX[i] = (L*dRdX[i] + R*dLdX[i])*s2psi + (P*dSdX[i] + S*dPdX[i])*(1.0 + c2psi);
393  dCdX[i] = P*R*dLdX[i] + P*L*dRdX[i] + R*L*dPdX[i];
394  dFdX[i] = 2.0*(dAdX[i] - dBdX[i] + dCdX[i]);
395  dGdX[i] = 2.0*dAdX[i] - dBdX[i] + (smu/J)*(B*dBdX[i] - 2.0*(A*dCdX[i] + C*dAdX[i]));
396  dmudX[i] = (0.5/my_mu.mu)*(dHdF*dFdX[i] + dHdG*dGdX[i]);
397 
398  dRdY[i] = X[i]/( pow(1.0 + Y[i], 2) );
399  dLdY[i] = -X[i]/( pow(1.0 - Y[i], 2) );
400  dSdY[i] = 0.5*(dRdY[i] + dLdY[i]);
401 
402  dAdY[i] = dSdY[i]*s2psi;
403  dBdY[i] = (L*dRdY[i] + R*dLdY[i])*s2psi + P*dSdY[i]*(1.0 + c2psi);
404  dCdY[i] = P*(L*dRdY[i] + R*dLdY[i]);
405 
406  dFdY[i] = 2.0*(dAdY[i] - dBdY[i] + dCdY[i]);
407  dGdY[i] = 2.0*dAdY[i] - dBdY[i] + (smu/J)*(B*dBdY[i] - 2.0*(A*dCdY[i] + C*dAdY[i]));
408 
409  dmudY[i] = (0.5/my_mu.mu)*(dHdF*dFdY[i] + dHdG*dGdY[i]);
410 
411 
412  dXdw[i] = -2.0*wp2[i]/w3;
413  dYdw[i] = -wc[i]/w2;
414 
415  }
416 
417  dAdpsi = sin(2.0*psi)*(S - P);
418  dBdpsi = sin(2.0*psi)*(R*L - P*S);
419  dFdpsi = 2.0*(dAdpsi - dBdpsi);
420  dGdpsi = 2.0*dAdpsi - dBdpsi + (smu/J)*(B*dBdpsi - 2.0*C*dAdpsi);
421 
422  my_mu.dmudtheta = (0.5/my_mu.mu)*(dHdF*dFdpsi + dHdG*dGdpsi);
423 
424  dmudw = 0.0;
425  for(int i=0; i<ncomps; i++) dmudw += dmudX[i]*dXdw[i] + dmudY[i]*dYdw[i];
426 
427  my_mu.dmudom = dmudw;
428  my_mu.cone_ang = std::atan(std::sqrt(-P/S));
429 
430  if(!skip_phi){
431  calc_type term1, term2, term3, denom, tmp_bes, tmp_besp, tmp_besm, bessel_arg, D_mu2S, omega_n;
432 
433  D_mu2S = D / (mu2 - S);
434  calc_type calc_n = (calc_type) n;
435  omega_n = -1.0 * calc_n * std::abs(get_omega_ref("ce"))/gamma_particle;
436  calc_type omega_n_slash_n = -1.0 * std::abs(get_omega_ref("ce"))/gamma_particle;
437  //temporaries for simplicity
438 
439  term1 = (mu2* s2psi - P)/(mu2);
440 
441  denom = std::pow(D_mu2S*term1, 2) + c2psi*std::pow((P/mu2), 2);
442 
443  bessel_arg = std::tan(psi)*std::tan(alpha) * (w - omega_n)/omega_n_slash_n;// n x tan alpha (om - om_n)/om_n, but allowing n/n = 1 even when n = 0
444 
445  tmp_besp = boost::math::cyl_bessel_j(n+1, bessel_arg);
446  tmp_besm = boost::math::cyl_bessel_j(n-1, bessel_arg);
447 
448  term2 = (1.0 + D_mu2S)*tmp_besp;
449  term2 += (1.0 - D_mu2S)*tmp_besm;
450 
451  if(n != 0) tmp_bes = 0.5*bessel_arg *(tmp_besp + tmp_besm)/calc_n;
452  else tmp_bes = boost::math::cyl_bessel_j(0, bessel_arg);
453  //Use bessel identity to save time.
454  // J_(n-1) + J_(n+1) = (2 n / arg) J_n, n!=0
455  term3 = scpsi*tmp_bes/std::tan(alpha);
456 
457  my_mu.phi = std::pow((0.5*term1*term2 + term3), 2)/denom;
458  if(alpha == 0) my_mu.phi = 0;
459  }
460  }
461  return my_mu;
462 }

464 std::vector<calc_type> plasma::get_resonant_omega(calc_type theta, calc_type v_par, calc_type gamma_particle, int n)const{
478 #ifdef DEBUG_ALL
479  //Argument preconditions. Check only in debug mode for speed
480  if(std::abs(v_par) >= v0) my_error_print("!!!!!!!!Error in get_resonant_omega, velocity (v_par="+mk_str(v_par)+") out of range!!!!!!", mpi_info.rank);
481  if(gamma_particle < 1.0) my_error_print("!!!!!!!!Error in get_resonant_omega, particle gamma (gamma_particle="+mk_str(gamma_particle)+") out of range!!!!!!", mpi_info.rank);
482 #endif

484  std::vector<calc_type> ret_vec;
485  calc_type om_ce = this->get_omega_ref("ce");
486  calc_type om_pe_loc = this->get_omega_ref("pe");
487  calc_type om_ce_ref = this->get_omega_ref("c0");
488  calc_type signed_om_ce_slash_ref = - std::abs(om_ce/om_ce_ref);//Contains sign of electron charge
489  //Special case when v=0 and we can save time
490  if(std::abs(v_par) < tiny_calc_type){
491  if( n == 1 || n == -1 ) ret_vec.push_back(om_ce*n);
492  return ret_vec;
493  }
494  //We let n=0 fall through even though we can handle it quicker because it's simpler code. We do exclude omega "=" zero solutions below though.
495 
496  //Equation to solve is cubic of the form
497  //a x^3 + b x^2 + c x + d = 0
498  //or equivalently
499  //x^3 + an x^2 + bn x + cn = 0
500 
501  calc_type a, b, c, d;
502  calc_type an, bn, cn;
503  calc_type cos_th = std::cos(theta);
504  calc_type vel = v_par / v0;
505  calc_type calc_n = (calc_type) n;
506  //For clarity
507  calc_type vel_cos = std::pow(vel * cos_th, 2);//Product of the two, for simplicity in expressions. But note is not meaningful, theta is the wave angle
508 
509  calc_type gamma_sq = gamma_particle * gamma_particle;
510 
511  //Calculate coefficients
512  //To maintain best precision we solve for x = omega/omega_ce_ref so x ~ 1
513  a = (vel_cos - 1.0) * gamma_sq;
514 
515  b = (vel_cos*cos_th*gamma_sq + 2.0*gamma_particle* calc_n - gamma_sq*cos_th)*signed_om_ce_slash_ref;
516 
517  c = (2.0 * gamma_particle * calc_n * cos_th - calc_n * calc_n)* std::pow(signed_om_ce_slash_ref, 2) - std::pow(om_pe_loc/om_ce_ref, 2)*vel_cos*gamma_sq;
518 
519  d = -calc_n*calc_n*std::pow(signed_om_ce_slash_ref, 3)*cos_th;
520  //Note: for n=0, d is 0. We covered v_par being zero above, so either omega = 0 and k_par = 0 or we have a normal solution, but with redundancy. We assume omega = 0 is an unhelpful solution to return. And for simplicity we don't do a special quadratic solution but just continue anyway
521 
522  an = b/a;
523  bn = c/a;
524  cn = d/a;
525 
526  ret_vec = cubic_solve(an, bn, cn);
527 
528  //Restore om_ce_ref factor and delete any entries > om_ce as these aren't whistler modes. Also delete answers which are "zero"
529  for(size_t i=0; i<ret_vec.size(); ++i) ret_vec[i] *= om_ce_ref;
530  for(size_t i=0; i<ret_vec.size(); ++i){
531  //We could remove those outside the NR_[min/max]_om bounds, but then we can't use these as a base for the NR version. But we don't expect solutions to move through 0 or om_ce
532  if(std::abs(ret_vec[i]) > std::abs(om_ce)*NR_max_om || std::abs(ret_vec[i]) < GEN_PRECISION){
533  ret_vec.erase(ret_vec.begin() + i);
534  --i;
535  }
536  }
537 
538  return ret_vec;
539

632  --i;
633  }
634  }

640 bool plasma::check_resonant_omega(calc_type theta, calc_type v_par, calc_type gamma_particle, int n, calc_type omega, calc_type & result)const{
641 
642  calc_type om_ce = this->get_omega_ref("ce");
643  calc_type om_pe_loc = this->get_omega_ref("pe");
644  calc_type om_ce_ref = this->get_omega_ref("c0");
645  calc_type signed_om_ce_slash_ref = - std::abs(om_ce/om_ce_ref);//Contains sign of electron charge
646 
647  calc_type a, b, c, d;
648  calc_type an, bn, cn;
649  calc_type cos_th = std::cos(theta);
650  calc_type vel = v_par / v0;
651  calc_type calc_n = (calc_type) n;
652  //For clarity
653  calc_type vel_cos = std::pow(vel * cos_th, 2);//Product of the two, for simplicity in expressions. But note is not meaningful, theta is the wave angle
654 
655  calc_type gamma_sq = gamma_particle * gamma_particle;
656 
657  //Calculate coefficients
658  //To maintain best precision we solve for x = omega/omega_ce_ref so x ~ 1
659  a = (vel_cos - 1.0) * gamma_sq;
660 
661  b = (vel_cos*cos_th*gamma_sq + 2.0*gamma_particle* calc_n - gamma_sq*cos_th)*signed_om_ce_slash_ref;
662 
663  c = (2.0 * gamma_particle * calc_n * cos_th - calc_n * calc_n)* std::pow(signed_om_ce_slash_ref, 2) - std::pow(om_pe_loc/om_ce_ref, 2)*vel_cos*gamma_sq;
664 
665  d = -calc_n*calc_n*std::pow(signed_om_ce_slash_ref, 3)*cos_th;
666  //Note: for n=0, d is 0. We covered v_par being zero above, so either omega = 0 and k_par = 0 or we have a normal solution, but with redundancy. We assume omega = 0 is an unhelpful solution to return. And for simplicity we don't do a special quadratic solution but just continue anyway
667 
668  an = b/a;
669  bn = c/a;
670  cn = d/a;
671 
672  result = std::pow(omega, 3) + an * std::pow(omega, 2) + bn * omega + cn;
673  calc_type res1, res2;
674  res1 = std::pow(omega*(1.0 - GEN_PRECISION), 3) + an * std::pow(omega*(1.0 - GEN_PRECISION), 2) + bn * omega*(1.0 - GEN_PRECISION) + cn;
675  res2 = std::pow(omega*(1.0 + GEN_PRECISION), 3) + an * std::pow(omega*(1.0 + GEN_PRECISION), 2) + bn * omega*(1.0 + GEN_PRECISION) + cn;
676  return (res1/res2 < 0);
677 
678 }
