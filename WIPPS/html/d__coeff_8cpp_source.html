164 d_report diffusion_coeff::calculate(D_type_spec type_of_D, bool quiet){
175 //----Initialise d_report --------
176  d_report report = {true, 0, 0, 0, false};
177 
178 //----- Get the plasma and spectrum bits
179  plasma plas;
180  spectrum * spect;
181  if(my_controller){
182  plas = my_controller->get_plasma();
183  spect = my_controller->get_current_spectrum();
184  }
185  else{
186  my_error_print("No controller", mpi_info.rank);
187  return report;
188  }
189  this->copy_ids(spect);//Copy block id, ranges etc from spect.
190  calc_type k_thresh = spect->check_upper();
191  my_print("Using omega upper threshold of "+ mk_str(plas.get_dispersion(k_thresh, WAVE_WHISTLER)/plas.get_omega_ref("ce")), 1);
192 
193 //---- Set type of D to calculate
194  bool is_mixed = false, is_pp = false;
195  if(type_of_D == D_type_spec::alpha_p || type_of_D == D_type_spec::p_alpha) is_mixed = true;
196  else if(type_of_D == D_type_spec::p_p) is_pp = true;
197 
198 //----- Major definitions------------------
199  //Velocity and angle temporaries
200  calc_type tan_theta, theta, c2th, mod_v, gamma_particle, alpha, s2alpha, cos_alpha;
201  //Temporaries for steps of D calculation
202  calc_type D_part_n_sum, D_final_without_consts, D_conversion_factor, D_part_n_contrib;
203  //More temporaries for parts of D
204  calc_type Eq6, numerator;
205  //Resonant freq and array of all frequency solutions
206  calc_type omega_n = 0.0;
207  std::vector<calc_type> omega_calc, omega_calc_tmp;
208  //Resonant numbers
209  int n_min, n_max;
210  //Plasma dispersion
211  mu_dmudom my_mu;
212 
213  calc_type om_ce_ref = std::abs(plas.get_omega_ref("ce")), omega_solution;
214 
215  calc_type D_consts = 0.25* pi*om_ce_ref*om_ce_ref*v0 * std::pow(1.0/plas.get_B0(), 2) * spect->get_norm_B();//Constant part of D ( pi/2 Om_ce*c^3) * m^2 / B_0^2. But we then divide by (m^2 c^2) so that we get D/p^2 by just dividing by (gamma^2 - 1) which is done in the last step
216  //Extra half I can't quite explain, but matches the sample data
217 
218 //-------- Wave angle temporaries -----------------
219  //D has to be integrated over x, so we need a temporary and the axis
220  calc_type *D_tan_theta = (calc_type *) calloc(n_thetas, sizeof(calc_type));
221  calc_type *x = (calc_type *) calloc(n_thetas, sizeof(calc_type));
222  calc_type *dx = (calc_type *) calloc(n_thetas, sizeof(calc_type));
223  for(int i = 0; i < n_thetas; ++i) x[i] = i* TAN_MAX/n_thetas;
224  for(int i = 0; i < n_thetas - 1; ++i) dx[i] = x[i+1] - x[i];
225 
226 //-------- Initialise reporting info --------
227  running_report running_rept = {0, 0, false};
228  first_running_report(dims[0], running_rept, quiet);
229  size_t n_av = 0, n_omega_solutions = 0;
230 
231 //------------------Main loops here-------------------------
232 //We have deep nested loops. Move ANYTHING that can be as far up tree as possible
233 
234 int good = 0, bad = 0;
235 
236  for(size_t v_ind = 0; v_ind < dims[0]; v_ind++){
237  //particle parallel velocity or momentum
238  mod_v = get_axis_element(0, v_ind);
239  gamma_particle = gamma_rel(mod_v);
240  if(std::abs(mod_v) < 1.0) continue;//Skip if velocity is tiny
241  //Print the progress info
242  do_running_report(v_ind, mod_v, report.n_min, report.n_max, running_rept);
243 
244  for(size_t part_pitch_ind = 0; part_pitch_ind < dims[1]; part_pitch_ind++){
245  //particle pitch angle
246  alpha = get_axis_element_ang(part_pitch_ind);
247  cos_alpha = std::cos(alpha);
248  s2alpha = 1.0 - std::pow(cos_alpha, 2);//s^2 + c^2 = 1
249  if(single_n){
250  //Use only the single selected resonance
251  n_min = n_used;
252  n_max = n_used;
253  }else{
254  //Get limits on n for this velocity and angle
255  n_min = get_min_n(mod_v, cos_alpha, k_thresh, om_ce_ref);
256  n_max = get_max_n(mod_v, cos_alpha, k_thresh, om_ce_ref);
257  n_av += n_max;//Track the average n_max over all iterations
258  report.n_max = std::max(report.n_max, (size_t)n_max);//Track the extreme n_max and n_min
259  report.n_min = std::max(report.n_min, (size_t) -n_min);
260  }
261  for(int wave_ang_ind = 0; wave_ang_ind < n_thetas; wave_ang_ind++){
262  //theta loop for wave angle or x=tan theta
263  tan_theta = x[wave_ang_ind];
264  theta = std::atan(tan_theta);
265  c2th = std::pow(std::cos(theta), 2);
266  D_part_n_sum = 0.0;
267 
268  for(int n = n_min; n <= n_max; ++n){
269  // n is resonant number
270  omega_calc_tmp = plas.get_resonant_omega(theta, mod_v*cos_alpha, gamma_particle, (calc_type) n);
271  omega_calc = plas.get_resonant_omega_full(theta, mod_v*cos_alpha, gamma_particle, (calc_type) n);
272 
273  if( omega_calc.size() != omega_calc_tmp.size()){
274  //std::cout<<omega_calc.size()<<' '<<omega_calc_tmp.size()<<" ; ";
275  bad ++;
276 // if(n != 0) std::cout<<n<<' ';
277  }
278  good ++;
279 
280  omega_n = - (calc_type) n * om_ce_ref/gamma_particle;
281  n_omega_solutions = omega_calc.size();
282 
283  for(size_t om_solution_num = 0; om_solution_num < n_omega_solutions; ++om_solution_num){
284  //Loop over solutions
286  omega_solution = omega_calc[om_solution_num];
287  //my_mu = plas.get_high_dens_phi_mu_om(omega_solution, theta, alpha, n, gamma_particle);
288  my_mu = plas.get_phi_mu_om(omega_solution, theta, alpha, n, gamma_particle);
289 
290 #ifdef DEBUG_ALL
291  //Sanity checking...
292  if(my_mu.cone_ang < theta){
293  my_error_print("Solution outside expected resonance cone, error! ", mpi_info.rank);
294  //throw std::domain_error("Solution outside resonance cone");
295  continue;
296  }
297 #endif
298  Eq6 = omega_solution/(omega_solution - omega_n)* my_mu.mu/(my_mu.mu + omega_solution * my_mu.dmudom);
299  numerator = std::pow( -s2alpha + omega_n/omega_solution, 2);
300  D_conversion_factor = sin(alpha)*cos(alpha)/(-s2alpha + omega_n/omega_solution);
301  //Get the part of D summed over n. Note additional factors below
302  //NB NB get_G_1 precancels the \Delta\omega and B_wave factors
303  D_part_n_contrib = numerator * my_mu.phi / std::abs(1.0 - Eq6) * get_G1(spect, omega_solution) * get_G2(spect, omega_solution, tan_theta);
304 
305  //Convert alpha_alpha to requested D type
306  D_part_n_contrib = (is_mixed ? D_part_n_contrib*D_conversion_factor : D_part_n_contrib);
307  D_part_n_contrib = (is_pp ? D_part_n_contrib*D_conversion_factor*D_conversion_factor : D_part_n_contrib);
308 
309  D_part_n_sum += D_part_n_contrib;
310  }
311  }
312  //Store into temporary theta array
313  D_tan_theta[wave_ang_ind] = D_part_n_sum * c2th * tan_theta;
314  }
315  //now integrate in x = tan theta, restore the velocity factor and save
316  D_final_without_consts = integrator(D_tan_theta, n_thetas, dx);
317 
318  D_final_without_consts /= mod_v*std::pow(cos_alpha, 3) * (gamma_particle*gamma_particle - 1.0);
319  set_element(v_ind, part_pitch_ind, D_final_without_consts*D_consts);
320  }
321  }
322 //------------End main loops---------------------------------
323 
324  free(dx);
325  free(x);
326  free(D_tan_theta);
327 
328 //----Assemble the final report -----------------------------
329  report.error = false;
330  report.n_av = n_av/dims[0]/dims[1];
331  if(single_n){
332  report.n_av = n_used;
333  report.n_max = 0;
334  report.n_min = 0;
335  report.single_n = true;
336  }
337 
338 std::cout<<good<<' '<<bad<<'\n';
339 
340  return report;
341 
342 }
