231 calc_type get_growth_rate(plasma * my_plas, non_thermal * my_elec, int n_momenta, calc_type * p_axis, calc_type omega_in){
243  omega_in = std::abs(omega_in);
244  calc_type k = my_plas->get_dispersion(omega_in, WAVE_WHISTLER, 1);
245 
246  calc_type ck_om = v0*k/omega_in, om_ce = std::abs(my_plas->get_omega_ref("ce")), om_pe = std::abs(my_plas->get_omega_ref("pe")), om_diff = omega_in - om_ce;
247  //This corrects omega_pe to match the fast electrons we're using, which may contain extra components
248  om_pe = om_pe*std::sqrt(my_elec->get_total_dens()/my_const.dens_factor);
249 
250  calc_type S_tot=0.0, S_full_tot=0.0, A_crit, A_rel, eta_rel;
251  calc_type *S, *S_full, * dp_ax;
252  calc_type gamma, p_res, Delta_res, v_res;
253  bool norely = my_elec->get_norely();
254 
255  S = (calc_type *)malloc(n_momenta*sizeof(calc_type));
256  S_full = (calc_type *)malloc(n_momenta*sizeof(calc_type));
257  dp_ax = (calc_type *)malloc(n_momenta*sizeof(calc_type));
258 
259  v_res = om_diff/k;
260 
261  for(int j=0; j< n_momenta; ++j){
262 
263  gamma = - 1.0 + ck_om * std::sqrt( (ck_om*ck_om -1.0 )*(1.0 + p_axis[j]*p_axis[j]/v0/v0)*(omega_in*omega_in/om_ce/om_ce) + 1.0 );
264  gamma /= ((ck_om*ck_om - 1.0)*omega_in/om_ce);
265  //14 in Xiao resonant gamma factor
266 
267  p_res = (gamma * omega_in - om_ce)/k;
268  //Resonant momentum
269  Delta_res = 1.0 - (omega_in*p_res / (v0*v0*k*gamma));
270  //Xiao 15, no meaning given. Always +ve
271  if(Delta_res < GEN_PRECISION) my_print("ERROR!! Delta is 0 or negative", mpi_info.rank);
272 
273  //Now p_par = p_res and p_perp is p_axis[j]
274  if(!norely){
275  // Relativistic version---------------------------------
276  S[j] = std::pow(p_axis[j], 2) * my_elec->d_f_p(p_res, p_axis[j], 0) / Delta_res;
277 
278  S_full[j] = std::pow(p_axis[j], 2)/ Delta_res/gamma * (p_axis[j]*my_elec->d_f_p(p_res, p_axis[j], 1) - p_res*my_elec->d_f_p(p_res, p_axis[j], 0));
279  }else{
280  //---Non-rely version--------------------------
281 
282  S[j] = p_axis[j]*my_elec->f_p(v_res, p_axis[j]);
283 
284  S_full[j] = - std::pow(p_axis[j], 2)* (p_axis[j]*my_elec->d_f_p(v_res, p_axis[j], 1) - v_res*my_elec->d_f_p(v_res, p_axis[j], 0));
285  }
286  }
287 
288  dp_ax[0] = 0.0;
289  for(int j=1; j<n_momenta; j++) dp_ax[j] = p_axis[j] - p_axis[j-1];
290 
291  S_tot = integrator(S, n_momenta, dp_ax);
292  S_full_tot = integrator(S_full, n_momenta, dp_ax);
293 
294  calc_type ret = 0.0;
295 
296  if(std::abs(S_tot) > std::numeric_limits<calc_type>::min()){
297  A_crit = - omega_in /om_diff;
298 
299  if(!norely){
300  // Relativistic version---------------------------------
301  A_rel = k*S_full_tot/S_tot/om_diff;
302  eta_rel = pi * om_diff/k * S_tot;
303  }else{
304  //---Non-rely version--------------------------
305  A_rel = S_full_tot/S_tot/2.0/v_res;
306  eta_rel = -2.0*pi * v_res * S_tot;
307  }
308 
309  ret = pi*om_pe*om_pe/(2.0*omega_in + om_pe*om_pe*om_ce/(std::pow(om_diff, 2))) * eta_rel * (A_rel - A_crit);
310  }
311 
312  free(S);
313  free(S_full);
314  free(dp_ax);
315 
316  return ret;
317 }

320 calc_type * make_momentum_axis(int n_mom, calc_type v_max){
329  //Momentum step size, including gamma
330  calc_type dp = v_max * v0/ (calc_type) (n_mom-1)/std::sqrt(1.0 - v_max*v_max);
331 
332  //Allocate memory
333  calc_type * p_axis;
334  p_axis = (calc_type*) malloc(n_mom*sizeof(calc_type));
335  if(!p_axis) return nullptr;
336 
337  p_axis[0] = 0.0;
338  //Set momenta from 0 to v_max.
339  for(int i = 1; i< n_mom; ++i) p_axis[i] = p_axis[i-1] + dp;
340  //Check for clipping errors in dp
341  if(std::abs(p_axis[n_mom-1] - (v_max * v0/ std::sqrt(1.0 - v_max*v_max))) > 1e-3) my_error_print("Momentum axis error of "+mk_str(std::abs(p_axis[n_mom-1] - (v_max * v0/ std::sqrt(1.0 - v_max*v_max)))), mpi_info.rank);
342 
343  return p_axis;
