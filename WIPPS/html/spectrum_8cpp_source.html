310 bool spectrum::make_angle_distrib(my_type std_dev){
319  if(!g_is_angle_only){
320  my_error_print("Angular distrib is not a function. Returning", mpi_info.rank);
321  return 1;
322  }
323  if(!g_angle_array.is_good()){
324  my_error_print("Angular array invalid. Returning", mpi_info.rank);
325  return 1;
326  }
327  make_angle_axis();
328  size_t len = get_angle_length();
329  my_type val;
330 
331  if(function_type == FUNCTION_DELTA){
332 
333  for(size_t i=0; i<len; ++i) set_g_element(i,0.0);
334  val = 1.0/(get_ang_axis_element(1)-get_ang_axis_element(0));
335  int zero = where_angle(0.0);
336  //Set_element checks bnds automatically
337  set_g_element(zero, val);
338 
339  }else if(function_type == FUNCTION_GAUSS){
340  my_type ax_el;
341  my_type norm;
342  norm = 1.0/ (std::sqrt(2.0*pi) * std_dev);
343  for(size_t i=0; i<len; ++i){
344  ax_el = get_ang_axis_element(i);
345  val = std::exp( -0.5 * std::pow(ax_el/std_dev, 2)) * norm;
346  set_g_element(i,val);
347  }
348  }else if(function_type == FUNCTION_ISO){
349 
350  val = 1.0/ (TAN_MAX - TAN_MIN)*get_angle_length()/(get_angle_length()-1);
351  if(std::abs(TAN_MIN) < GEN_PRECISION) val = val / 2.0;
352  for(size_t i=0; i<len; ++i) set_g_element(i,val);
353 
354  }else{
355 
356  my_error_print("Invalid function type. Returning", mpi_info.rank);
357  return 1;
358 
359  }
360 
361  return 0;

432 bool spectrum::generate_spectrum(data_array &parent, int om_fuzz, int angle_type, my_type std_dev, data_array * mask){
445  if(!this->is_good()){
446  my_error_print("Spectrum object invalid. Returning", mpi_info.rank);
447  return 1;
448  }
449  if(parent.get_dims() == 0 || parent.get_dims(0) != this->get_omega_length()){
450  my_error_print("0-dim of parent must match omega_length", mpi_info.rank);
451  return 1;
452  }
453 
454 
455  if(parent.is_good() && g_is_angle_only){
456  //Require parent to be 2-D
457  if(parent.get_dims() != 2){
458  my_error_print("Functional angle form requires 2-D parent", mpi_info.rank);
459  return 1;
460  }
461 
462  //First we read ids from parent
463  this->copy_ids(parent);
464  this->set_extra_ids(WAVE_WHISTLER, angle_type);
465  //...and omega axis
466  size_t len;
467  const my_type * om_ax = parent.get_axis(1, len);
468 
469  //Now we loop across x, calculate the wave cutout bounds, and total, putting result into data
470 
471  int j;
472  int low_bnd, high_bnd;
473  my_type om_disp, max=0.0;
474  my_type tolerance = om_fuzz/100.0;
475  my_type total = 0.0;
476 
477  calc_type sgn;
478  //One sided axis (runs 0 to max) rather than -max to max. These are the only options
479  bool one_sided = (std::abs(parent.get_axis_element(1,0)) < std::abs(parent.get_axis_element(1,len/2)));
480 
481  for(size_t i = 0; i < get_omega_length(); ++i){
482  //Convert parent k-axis value to omega, preserving sign
483  om_disp = get_omega(parent.get_axis_element(0,i), WAVE_WHISTLER);
484  if(!one_sided && i<= get_omega_length()/2) sgn = -1.0;
485  else sgn = 1.0;
486  om_disp *= sgn;
487  //Set spectrum axis to this value
488  set_om_axis_element(i, om_disp);
489 
490  //Lookup the parent omega axis values for both ends of band
491  low_bnd = where(om_ax, len, om_disp *(1.0-sgn*tolerance));
492  high_bnd = where(om_ax, len, om_disp *(1.0+sgn*tolerance));
493 
494  //Adjust for low off bottom or high off top
495  //Low off top or high off bottom will still be caught and skipped
496  if(low_bnd < 0 && om_disp *(1.0-sgn*tolerance) < parent.get_axis_element(1,0)) low_bnd = 0;
497  if(high_bnd < 0 && om_disp *(1.0+sgn*tolerance) > parent.get_axis_element(1,len-1)) high_bnd = len-1;
498 
499  if(low_bnd < 0 || high_bnd< 0){
500  set_B_element(i,0.0);
501  continue;
502  }
503  //now total the part of the array between these bnds
504  total=0.0;
505  for(j=low_bnd; j<high_bnd; j++) total += parent.get_element(i,j);
506  if(mask) for(j=low_bnd; j<high_bnd; j++) mask->set_element(i, j, mask->get_element(i, j)+1);
507 
508  set_B_element(i,total);
509  if(total > max) max = total;
510  }
511  //Generate the angular distrib function
512  make_angle_distrib(std_dev);
513  this->max_power = max;
514 
515  }else if(parent.is_good() && parent.get_dims()==3){
516  //This will do a rough integral around the correct omega. Note it is not a neat sampling of the domain, so may give odd results for non-smooth cases
517 
518  this->copy_ids(parent);
519  this->set_extra_ids(WAVE_WHISTLER, FUNCTION_NULL);
520 
521  //Convert om_fuzz from percentage to fraction
522  my_type tolerance = om_fuzz/100.0;
523 
524  //Get the k_x and k_y axes for lookup
525  size_t len_x, len_y, len_om;
526  const my_type * kx_ax = parent.get_axis(0, len_x);
527  const my_type * ky_ax = parent.get_axis(1, len_y);
528  const my_type * om_ax = parent.get_axis(2, len_om);
529 
530  //One sided axis (runs 0 to max) rather than -max to max. These are the only options
531  bool one_sided = (std::abs(om_ax[0]) < std::abs(om_ax[len_om/2]));
532 
533  //Create angular axis for spectrum
534  make_angle_axis();
535 
536  my_type k, tantheta, tmp, tmp_sum, max_power=0.0, theta, om_disp;
537  int kx_high, ky_low, ky_high, i_sgn=1, om_low, om_high;
538 
539  //Now we do a double loop over k_x and angle
540  for(size_t i = 0; i< len_x; i++){
541  //Signs so we always integrate from k_low to k_high, if one sided is always 1
542  if(!one_sided) i_sgn = (i<len_x/2 ? -1: 1);
543 
544  //Use the k_x axis to generate an omega one. Thus if fuzz=0 we match 1-1
545  om_disp = get_omega(parent.get_axis_element(0,i), WAVE_WHISTLER);
546  set_om_axis_element(i, om_disp*i_sgn);
547 
548  //We can assume either +ve and -ve k (include i_sgn in kx_high, or omega (include in om_low and high)
549 
550  tmp_sum = 0.0;
551  for(size_t j = 0; j< get_angle_length(); j++){
552  tantheta = get_ang_axis_element(j);
553  theta = atan(tantheta);
554 //----------Fuzz omega--------------------
555  om_low = where(om_ax, len_om, om_disp*(1.0-i_sgn*tolerance)*i_sgn);
556  om_high = where(om_ax, len_om, om_disp*(1.0+i_sgn*tolerance)*i_sgn);
557  if(om_low < 0) om_low = 0;
558  if(om_disp*(1.0+tolerance) < *om_ax) om_high = 0;
559  else if(om_high < 0) om_high = len_om - 1;
560 
561 //-------No fuzz in k_x -----------------------
562  k = get_k(om_disp, WAVE_WHISTLER, 0, theta);
563 
564  kx_high = where(kx_ax, len_x, k*cos(theta));
565 
566  if(k*cos(theta)< *(kx_ax) || std::abs(k)<GEN_PRECISION || k*cos(theta)> *(kx_ax+len_x -1)) kx_high = -1;
567  else if(kx_high <0) kx_high = len_x-1;
568 
569 //-------- k_y is fuzzy----------
570  k = get_k(om_disp*(1.0+tolerance), WAVE_WHISTLER, 0, theta);
571  ky_high = where(ky_ax, len_y, k*sin(theta));
572  if(std::abs(k)<GEN_PRECISION || std::abs(k*sin(theta))> *(ky_ax+len_y -1)) ky_high = -1;
573  else if(ky_high <0) ky_high = len_y-1;
574 
575  k = get_k(om_disp*(1.0-tolerance), WAVE_WHISTLER, 0, theta);
576  ky_low = where(ky_ax, len_y, k*sin(theta));
577  if(ky_low < 0) ky_low = 0;
578 
579 //-------Total on small range and create mask if requested-----
580  tmp = 0.0;
581  if(kx_high != -1 && ky_high != -1){
582  for(int ii=om_low; ii<=om_high; ii++){
583  for(int jj=ky_low; jj<=ky_high; jj++){
584  tmp += parent.get_element(kx_high, jj, ii);
585  }
586  if(!one_sided){
587  //Also include negative k_y
588  for(int jj=len_y-ky_high; jj<(long)len_y-ky_low; jj++){
589  tmp += parent.get_element(kx_high, jj, ii);
590  }
591  }
592  }
593  }
594  if(mask){
595  if(kx_high != -1 && ky_high != -1){
596  for(int ii=om_low; ii<=om_high; ii++){
597  for(int jj=ky_low; jj<=ky_high; jj++){
598  if(!one_sided) mask->set_element(kx_high, jj, ii, mask->get_element(kx_high, jj, ii)+0.5);
599  else mask->set_element(kx_high, jj, ii, mask->get_element(kx_high, jj, ii)+1.0);
600  }
601  if(!one_sided){
602  //Also include negative k_y
603  for(int jj=len_y-ky_high; jj<(long)len_y-ky_low; jj++){
604  mask->set_element(kx_high, jj, ii, mask->get_element(kx_high, jj, ii)+0.5);
605  }
606  }
607  }
608  }
609  }
610 //------------------
611  //Tmp is now the convolved B(omega)g(omega, theta). We dump it into g as is for later norming and add to the sum which will end up as B.
612  if(one_sided) tmp *= 2.0;//If one_sided axis, assume there's the same power at -ve omega too
613  set_g_element(i, j, tmp);
614  tmp_sum += tmp;
615  }
616  //Now the angles are done, we norm. g and set B
617  //Use some small value to prevent a divide by zero
618  if(std::abs(tmp_sum) > tiny_my_type){
619  for(size_t j = 0; j< get_angle_length(); j++) set_g_element(i, j, get_g_element(i, j)/tmp_sum);
620  }
621  set_B_element(i, tmp_sum);
622  //Store the max power for later
623  if(tmp_sum > max_power) max_power = tmp_sum;
624 
625  }
626  this->max_power = max_power;
627  }else{
628  return 1;
629  }
630 
631  //Correct the spectrum to be properly X^2(omega) dOmega which is what we ultimately want
632  this->convert_FFT_to_integral();
633 
634  //Calc and store normalising info etc
635  this->init();
636 
637  return 0;

835 bool spectrum::calc_norm_g(size_t om_ind){
843  size_t len = get_angle_length();
844  if(!my_controller) return 1;
845  plasma plas = my_controller->get_plasma();
846 
847  my_type * d_axis = (my_type *) calloc(len, sizeof(my_type));
848  my_type * integrand = (my_type *) calloc(len, sizeof(my_type));
849 
850  for(size_t i = 0; i < len-1; i++) d_axis[i] = get_ang_axis_element(i+1) - get_ang_axis_element(i);
851  //Construct dx axis for integration. Note x = tan theta
852

854  my_type x, psi, omega = std::abs(get_om_axis_element(om_ind)), g_el;
855

858  if(om_ind >= lena) return 1;

864  if(!my_mu.err){

866  g_el = get_g_element(i);
867  }else{
868  g_el = get_g_element(om_ind, i);
869  }

871 
872  }else{
873  integrand[i] = 0.0;
874  }
875  //product of g(theta) * x (x^2+1)^-(3/2) * mu^2 |mu+omega dmu/domega|
