157  if(n_tims > 0) spec_norm *= n_tims;
160  spec_norm *= spec_norm;

247  int i = 1;

249  if(strcmp(argv[i], "-f")==0 && i < argc-1){

284  my_error_print(std::string("UNKNOWN OPTION ")+argv[i], mpi_info.rank);
285  }

287  }

289  values.d[0] = MAX_SIZE;

299  //If using a file-list extract the names

301  values.file_list = process_filelist(argc, argv);
