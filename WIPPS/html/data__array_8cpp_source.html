728 bool data_array::resize(size_t dim, size_t sz, bool verbose){
738  size_t old_sz = this->get_dims(dim);
739  //Call my_array::resize to resize data portion
740  bool err = my_array::resize(dim, sz, verbose);
741 
742  if(!err){
743  //Success! Do axes!
744  my_type * new_ax;
745  size_t part_sz = 0;
746 
747  if(dim == n_dims-1){
748  //Special case as we can shrink and maybe grow without copy
749  //New size is sum of all other dims + sz
750  for(size_t i=0; i<dim; ++i) part_sz+= dims[i];
751  new_ax = (my_type *) realloc((void*) this->axes, (part_sz+sz)*sizeof(my_type));
752  if(!new_ax){
753  //Failure. Axes now don't match data array!
754  //This is bad and will do weird out of range stuff if left
755  my_error_print("Failed to reallocate axes", mpi_info.rank);
756  this->axes = nullptr;
757  return 1;
758  }
759  //Zero the newly allocated elements
760  long new_els = (sz - dims[dim]);
761  if(new_els > 0) memset((void*)(new_ax + part_sz), 0, new_els*sizeof(my_type));
762  //Set axes to point to new block
763  axes = new_ax;
764  }else{
765  //We have to allocate a new block and copy across
766  //New size is sum of all other dims + sz
767  for(size_t i=0; i<dim; ++i) part_sz += dims[i];
768  for(size_t i=dim+1; i<n_dims; ++i) part_sz += dims[i];
769 
770  //Allocate and copy
771  new_ax=(my_type*)calloc(part_sz+sz,sizeof(my_type));
772  size_t els_to_copy = 0, old_starts=0, new_starts=0;
773  for(size_t i=0; i<n_dims; ++i){
774  els_to_copy = dims[i];
775  if(i == dim && sz < old_sz) els_to_copy = sz;
776  std::copy(axes+old_starts, axes+old_starts+els_to_copy, new_ax+new_starts);
777  //Advance the offsets by correct amounts for old and new copies
778  if(i!= dim) old_starts += dims[i];
779  else old_starts +=old_sz;
780  new_starts +=els_to_copy;
781 
782  }
783  //Free old block and point axes to new block
784  free(axes);
785  axes = new_ax;
786 
787  }
788  return 0;
789  }else{
790  return 1;
791  }

795 bool data_array::shift(size_t dim, long n_els, bool axis){
804  if(dim >= n_dims) return 0;
805 
806  //my_array takes care of data shifting
807  bool err = my_array::shift(dim, n_els);
808 
809  if(axis && !err){
810  //Rotate the specified axis if we succeeded
811  //Adjust shift amount to be positive. This is always possible for cyclical shift
812  long sign_n = (n_els<0 ? -1: 1);
813  n_els = sign_n >0? (std::abs(n_els)%dims[dim]) : dims[dim]-(std::abs(n_els)%dims[dim]) ;
814  //By now n_els guaranteed >= 0
815 
816  size_t len=0;
817  const my_type * ax = get_axis(dim, len);
818  my_type * new_axis = (my_type*)malloc(len*sizeof(my_type));
819  if(!new_axis){
820  //Since the data shift succeeded, this should never happen in practise
821  my_error_print("Error allocating spare memory for shift", mpi_info.rank);
822  return 1;
823  }
824  long actual_shift = 0;
825  //Copy out axis, rotate, and copy back
826  std::copy(ax,ax+len, new_axis);
827  for(size_t j=0; j< len; j++){
828  //This is inefficient but is minimmaly changed from shift for whole array
829  actual_shift = (j+n_els >= dims[dim])? (j+n_els-dims[dim]): j+n_els;

1122 
1123  size_t num_seconds = data_out.get_dims() >= 2? data_out.get_dims(1): 1;//Size of "second dimension"

1128 
1129  for(size_t i=0; i< num_strides; i++){

1132  }
1133

1135  for(size_t j=0; j< num_strides/num_seconds; j++){
1136  std::reverse_copy(data_out.data+1+j*slice_sz, data_out.data+last_size+j*slice_sz, data_out.data+last_size-1+j*slice_sz);

1140  }

1145  shft = data_out.get_dims(i)/2;
